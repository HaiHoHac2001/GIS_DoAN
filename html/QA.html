10. Kiểm tra sông của các huyện

CREATE FUNCTION fun_check_inside_River_of_Huyen(@type character(20)) RETURNS TABLE AS RETURN (
SELECT id_poly, River.name, CASE Huyen.geom.STIntersects(River.geom) WHEN 1 THEN 'INSIDE' WHEN 0 THEN 'NOT INSIDE' END AS 'Checked'
FROM River, Huyen
WHERE Huyen.name = @type )
(nếu chỉ toàn là 'NOT INSIDE => huyện đó không có sông)
Kết quả: Huyện Krong Buk và Buon Ho
